C

Assembly Linux x86-64
section .data
    msg db "Hello World", 10 

section .text
    global _start

_start:
    mov rax, 1          ; write syscall
    mov rdi, 1          ; stdout
    mov rsi, msg        ; string pointer
    mov rdx, 14         ; string length
    syscall 

    mov rax, 60         ; exit syscall
    xor rdi, rdi        ; exit code 0
    syscall
Assembly Windows x86-64
extrn GetStdHandle : proc
extrn WriteFile : proc
extrn ExitProcess : proc

.data
msg db 'Hello World', 13, 10
msgLen_bytes equ $ - msg
written dq 0

.code
main proc
    sub rsp, 40          ; Shadow space
    mov ecx, -11         ; stdout handle
    call GetStdHandle    
    mov rbp, rax

    mov rcx, rbp         ; handle
    lea rdx, msg         ; buffer
    mov r8d, msgLen_bytes; len
    lea r9, written      ; var pointer
    mov qword ptr [rsp+32], 0
    call WriteFile       

    xor ecx, ecx         ; exit 0
    call ExitProcess    
main endp
end
